\newcommand{\includesvg}[1]
{
	\IfFileExists{#1.pdf}
	{
		\executeiffilenewer{#1.svg}{#1.pdf}
		{
			inkscape -z -D --file=#1.svg --export-pdf=#1.pdf --export-latex
		}
	}
	{
		\immediate\write18
		{
			inkscape -z -D --file=#1.svg --export-pdf=#1.pdf --export-latex
		}
	}
	\input{#1.pdf_tex}
}
